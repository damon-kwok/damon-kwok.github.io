#+BEGIN_SRC cpp
#include "StringHelper.h"
bool FileHelper::FileExists(FString Path) {
#ifdef UE4_EDITOR
    IPlatformFile &PlatformFile = FPlatformFileManager::Get().GetPlatformFile();
    return PlatformFile.FileExists(*Path);
#else
    std::fstream File_;
    File_.open(Path, std::ios::in);
    bool IsExist = false;
    if (File_)
        IsExist = true;

    File_.close();
    return IsExist;
#endif
}
